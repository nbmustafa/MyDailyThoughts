3. Backend (Go Proxy)

main.go

package main

import (
	"bytes"
	"io"
	"net/http"
	"os"
)

var apiKey = os.Getenv("ANTHROPIC_API_KEY")
var baseURL = os.Getenv("BASE_URL")

func proxy(w http.ResponseWriter, r *http.Request) {
	body, _ := io.ReadAll(r.Body)

	req, _ := http.NewRequest("POST", baseURL+"/v1/messages", bytes.NewBuffer(body))
	req.Header.Set("Content-Type", "application/json")
	req.Header.Set("x-api-key", apiKey)

	client := &http.Client{}
	resp, err := client.Do(req)
	if err != nil {
		http.Error(w, err.Error(), 500)
		return
	}
	defer resp.Body.Close()

	data, _ := io.ReadAll(resp.Body)
	w.WriteHeader(resp.StatusCode)
	w.Write(data)
}

func health(w http.ResponseWriter, r *http.Request) {
	w.Write([]byte("ok"))
}

func main() {
	http.HandleFunc("/api/analyze", proxy)
	http.HandleFunc("/healthz", health)
	http.ListenAndServe(":8080", nil)
}

6. Helm Templates

Frontend Deployment

apiVersion: apps/v1
kind: Deployment
metadata:
  name: frontend
spec:
  replicas: {{ .Values.replicaCount.frontend }}
  selector:
    matchLabels: { app: frontend }
  template:
    metadata:
      labels: { app: frontend }
    spec:
      nodeSelector:
        karpenter.sh/nodepool: {{ .Values.karpenter.nodePool }}

      topologySpreadConstraints:
        - maxSkew: 1
          topologyKey: topology.kubernetes.io/zone
          whenUnsatisfiable: ScheduleAnyway
          labelSelector:
            matchLabels:
              app: frontend

      containers:
        - name: frontend
          image: {{ .Values.image.frontend }}
          ports:
            - containerPort: 80

          resources:
            {{- toYaml .Values.resources.frontend | nindent 12 }}

          readinessProbe:
            httpGet: { path: "/", port: 80 }
            initialDelaySeconds: 5

          livenessProbe:
            httpGet: { path: "/", port: 80 }
            initialDelaySeconds: 15


---

Backend Deployment

apiVersion: apps/v1
kind: Deployment
metadata:
  name: backend
spec:
  replicas: {{ .Values.replicaCount.backend }}
  selector:
    matchLabels: { app: backend }
  template:
    metadata:
      labels: { app: backend }
    spec:
      serviceAccountName: backend-sa

      nodeSelector:
        karpenter.sh/nodepool: {{ .Values.karpenter.nodePool }}

      topologySpreadConstraints:
        - maxSkew: 1
          topologyKey: topology.kubernetes.io/zone
          whenUnsatisfiable: ScheduleAnyway
          labelSelector:
            matchLabels:
              app: backend

      containers:
        - name: backend
          image: {{ .Values.image.backend }}
          ports:
            - containerPort: 8080

          env:
            - name: ANTHROPIC_API_KEY
              valueFrom:
                secretKeyRef:
                  name: backend-secret
                  key: apiKey

              value: https://api.anthropic.com

          resources:
            {{- toYaml .Values.resources.backend | nindent 12 }}

          readinessProbe:
            httpGet: { path: "/healthz", port: 8080 }
            initialDelaySeconds: 5

          livenessProbe:
            httpGet: { path: "/healthz", port: 8080 }
            initialDelaySeconds: 10

    name: backend-secret
